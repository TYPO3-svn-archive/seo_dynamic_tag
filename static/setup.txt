plugin.tx_seodynamictag_pi1 {
  debug = {$plugin.tx_seodynamictag.debugging}
    // extensions
  templates =
  templates {
      // tt_news
    extensions =
    extensions {
        // metaTags, belowTheContent
      tt_news =
      tt_news {
          // author, description, keywords
        metaTags =
        metaTags {
            // field >, ifEmpty >, data
          author =
          author {
            field   >
            ifEmpty >
            data    = register:author
          }
            // field >, ifEmpty >, data
          description =
          description {
            field   >
            ifEmpty >
            data    = register:description
          }
            // field >, ifEmpty >, data
          keywords =
          keywords {
            field   >
            data    = register:keywords
            ifEmpty = {$plugin.tx_seodynamictag.phrases.keywords}
          }
        }
          // meta tag author, meta tag description, meta tag keywords, title 
        belowTheContent = COA
        belowTheContent {
            // meta tag author
          10 = USER
          10 {
            includeLibs = typo3conf/ext/seo_dynamic_tag/pi1/class.tx_seodynamictag_pi1.php
            userFunc    = tx_seodynamictag_pi1->main
            special   = register
            register  = author
            query {
              select    = `author`
              from      = `tt_news`
              where     = `uid` = $1 && `deleted` = 0 && `hidden` = 0
              var.1     = tx_ttnews[tt_news]
            }
            debug   = {$plugin.tx_seodynamictag.tt_news.debugging.author}
          }
            // meta tag description
          20 = USER
          20 {
            includeLibs = typo3conf/ext/seo_dynamic_tag/pi1/class.tx_seodynamictag_pi1.php
            userFunc    = tx_seodynamictag_pi1->main
            special   = register
            register  = description
            query {
              select    = `bodytext`
              from      = `tt_news`
              where     = `uid` = $1 && `deleted` = 0 && `hidden` = 0
              var.1     = tx_ttnews[tt_news]
              maxLength = 200
            }
            debug   = {$plugin.tx_seodynamictag.tt_news.debugging.description}
          }
            // meta tag keywords
          30 = USER
          30 {
            includeLibs = typo3conf/ext/seo_dynamic_tag/pi1/class.tx_seodynamictag_pi1.php
            userFunc    = tx_seodynamictag_pi1->main
            special   = register
            register  = keywords
            query {
              select    = `keywords`
              from      = `tt_news`
              where     = `uid` = $1 && `deleted` = 0 && `hidden` = 0
              var.1     = tx_ttnews[tt_news]
              keywords  = 0
            }
            debug   = {$plugin.tx_seodynamictag.tt_news.debugging.keywords}
          }
            // title
          40 = USER
          40 {
            includeLibs = typo3conf/ext/seo_dynamic_tag/pi1/class.tx_seodynamictag_pi1.php
            userFunc    = tx_seodynamictag_pi1->main
            special = title
            query {
              select  = CONCAT(`title`, ': ', `short`)
              from    = `tt_news`
              where   = `uid` = $1 && `deleted` = 0 && `hidden` = 0
              var.1   = tx_ttnews[tt_news]
            }
            debug   = {$plugin.tx_seodynamictag.tt_news.debugging.title}
          }
        }
      }
    }
  }
}
